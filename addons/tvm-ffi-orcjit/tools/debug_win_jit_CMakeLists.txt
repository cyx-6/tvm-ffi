cmake_minimum_required(VERSION 3.20)
project(debug_win_jit CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find LLVM
find_package(LLVM REQUIRED CONFIG)
message(STATUS "Found LLVM ${LLVM_PACKAGE_VERSION}")
message(STATUS "Using LLVMConfig.cmake in: ${LLVM_DIR}")

llvm_map_components_to_libnames(LLVM_LIBS
  Core
  OrcJIT
  Support
  native
)

add_executable(debug_win_jit debug_win_jit.cpp)
target_include_directories(debug_win_jit PRIVATE ${LLVM_INCLUDE_DIRS})
target_compile_definitions(debug_win_jit PRIVATE ${LLVM_DEFINITIONS})
target_link_libraries(debug_win_jit PRIVATE ${LLVM_LIBS})

if(WIN32)
  target_link_libraries(debug_win_jit PRIVATE ntdll psapi)
endif()
